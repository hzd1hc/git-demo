*******************************************************************************************************
*		command			|			description		   	      *
*******************************************************************************************************
cd					|  change directory
cd ../					|  back to upper directory
cd ../folder				|  move to folder with same level
ls					|  list all artifacts in current directory
git init				|  initialize .git
git version				|  get git version
git status				|  get staged files to commit
git log					|  get git history
git clone <url>				|  cloning files from remote repo
git add <files>				|  add files to commit
git commit -m <message>			|  commit the changes
git merge <branch>			|  merge changes from other branches to current branch
git push -u origin <branch>		|  push changes to remote repo
git pull				|  pull changes from remote repo to local repo
git remote add origin <link>		|  add changes to remote repo for the first time
git branch <name>			|  create new branch
git checkout <branch>			|  move to the branch
git remote -v				|  get remote repo which will store the changes
git diff				|  get difference between two branch
git reset				|  unstaged changes which not wanted to commit
git commit -am				|  combination of git add and git commit
git fetch <url>				|  to fetch the remote repo
git fetch <branch url> <branch_name>	|  to fetch a specific branch
git fetch -all				|  to fetch all branches
git fetch origin			|  to synchronize remote repo to local repo
